Infinitely growing list

                        
<ul data-bind="foreach: items, bottomsUp: true">
    <li data-bind="text: $data"></li>
</ul>
                    
                        
var viewModel = { items: ko.observableArray() },
    i = 1;
ko.applyBindings(viewModel);
setInterval(function () {
    viewModel.items.push('foo ' + i++);
}, 1000);
